PostCSS Plugins

Подключение через webpack.config.js

webpack.config.js
module: {
    rules: [
        {
            test: /\.scss/,
            use: [
                MiniCssExtractPlugin.loader,
                'css-loader',
                
                {
                    loader: 'postcss-loader',
                    options: {
                        plugins: [
                            //
                        ],
                        sourceMap: true
                    }
                },
                
                'sass-loader'
            ]
        },
    ]
}

Подключение через postcss.config.js

webpack.config.js
module: {
    rules: [
        {
            test: /\.scss/,
            use: [
                MiniCssExtractPlugin.loader,
                'css-loader',
                
                {
                    loader: 'postcss-loader',
                    options: {
                        config: { path: './postcss.config.js' },
                        sourceMap: true
                    }
                },
                
                'sass-loader'
            ]
        },
    ]
}

postcss.config.js
module.exports = {
    plugins: [
        require('autoprefixer'),
        require('css-mqpacker'),
        require('cssnano')
    ]
}
