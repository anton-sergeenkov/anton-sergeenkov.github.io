Русскоязычная документация 
bash
rmdir моя_папка # удаление пустую папку
rm имя_файла    # удаление файла
rm -R моя_папка # удаление папки с файлами (-r - опция рекурсивного удаления )


wget name_url                     # скачивание
sudo apt-get install name_program # скачивание с репозитория и установка
sudo apt-get remove name_program  # удаление пакета
sudo dpkg -i name_package         # установка скачанного пакета с расширением .deb
sudo apt-get update               # обновление
