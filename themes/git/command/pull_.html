CLONE / PULL


$ git clone git@github.com:anton-sergeenkov/repo.git # создание локальной копии удаленного git-репозитория (выполняется 1 раз)
$ git pull # получить изменения из удаленного git-репозитория в локальный репозиторий

pull


$ git pull origin        # Замерджить все ветки с удаленного репозитория
$ git pull origin master # Аналогично предыдущему, но накатывается только ветка master
$ git pull origin HEAD   # Накатить текущую ветку, не вводя ее длинное имя
