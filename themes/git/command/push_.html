INIT / PUSH ...


$ git init                  # инициализация пустого локального git-репозитория. Создаёт в текущем каталоге новый подкаталог с именем .git
$ git add .                 # добавление новых файлов из Working Directory в систему контроля версий (в индекс Staging Area для последующего коммита). Индексирование файлов. Сборка слепка коммита
$ git commit -m "text"      # создание слепка (snapshot) директории и зафиксировать изменения файлов
$ git remote add origin git@github.com:anton-sergeenkov/repo.git # добавление к локальному репозиторию ссылки на удаленный репозиторий
$ git push -u origin master # отправка изменений на удаленный git-репозиторий

$ push


$ git push --set-upstream origin master                 # отправка изменений на удаленный git-репозиторий
$ git push origin                                       # замерджить все ветки локального репозитория на удаленный репозиторий 
$ git push origin master                                # аналогично предыдущему, но делается пуш только ветки master
$ git push origin HEAD                                  # запушить текущую ветку, не вводя целиком ее название
$ git push -f                                           # залить на сервер измененные коммиты
$ git push https://github.com/Eclerrr/readme.git master #

$ add


$ git add .          # индексирование всех файлов
$ git add index.html # индексирование одного файлв

$ commit


$ git commit              # сохранение изменений в локальном репозитории. Берёт все данные, добавленные в индекс с помощью git add, и сохраняет их слепок во внутренней базе данных, а затем сдвигает указатель текущей ветки на этот слепок
$ git commit -a           # добавление всех изменений в индекс без использования git add
$ git commit -m "text"    # передача сообщения коммита без запуска полноценного редактора
$ git commit -a -m "text" #
$ git commit --amend -m "Перезапись коммита" # перезапишется последний коммит
