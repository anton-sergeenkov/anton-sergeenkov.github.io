const axios = require('axios');

GET


axios.get('url')
.then(response => {
	console.log(response.data);
})
.catch(error => console.log(error));

POST


axios.post('url', {
		firstName: 'Fred',
		lastName: 'Flintstone'
	})
	.then(function (response) {
		console.log(response);
	})
	.catch(function (error) {
		console.log(error);
	});
