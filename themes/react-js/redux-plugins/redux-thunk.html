redux/reducers/store.js
import { createStore, applyMiddleware } from 'redux';
import { composeWithDevTools } from 'redux-devtools-extension';
import thunk from 'redux-thunk';
import rootReducer from './reducers';

export default createStore(rootReducer, composeWithDevTools(applyMiddleware(thunk)));
  • Если action creator возвращает не простой объект, а функцию - она выполняется, иначе если это простой объект - передается дальше. Более того, в цепочке middleware есть доступный метод dispatch и getState
