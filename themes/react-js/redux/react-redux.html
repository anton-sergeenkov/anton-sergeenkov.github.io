Документация
  • Provider - рутовый компонент для того чтобы прокидывать Store. Необходим для того чтобы не пробрасывать Store в каждый компонент самостоятельно. store будет доступен в каждом дочернем компоненте Provider

import { createStore } from 'redux';       // создание store
import { Provider }    from 'react-redux'; // рутовый компонент для того чтобы прокидывать Store
import { connect }     from 'react-redux'; // декоратор
