  • Provider - рутовый компонент для того чтобы прокидывать Store. Необходим для того чтобы не пробрасывать Store в каждый компонент самостоятельно. store будет доступен в каждом дочернем компоненте Provider

import { createStore }     from "redux";       // создание store
import { Provider }        from "react-redux"; // рутовый компонент для того чтобы прокидывать Store
import { connect }         from "react-redux"; // function for you to connect your component to the store (декоратор)
import { combineReducers } from "redux";       // объединение нескольких reducer

connect()


connect()(Component)                                    // Subscribe=False Action=False
connect(mapStateToProps)(Component)                     // Subscribe=True  Action=False
connect(null, mapDispatchToProps)(Component)            // Subscribe=False Action=True
connect(mapStateToProps, mapDispatchToProps)(Component) // Subscribe=True  Action=True
