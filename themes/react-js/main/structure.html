Структура

  • Один компонент App, который находится на самом верху
  • Существует два типа «модели» данных: пропсы this.props и состояние this.state
Component App
class App extends React.Component {
	constructor
	constructor(props) {
		super(props);
		this.state = {
			name: 'Tony'
		};
	}
	

метод
	changeName(name) {
		// синхронно
		this.setState({
			name: name
		});
		// асинхронно
		this.setState((state, props) => ({
			info: state.name + props.age
		}));
	}
	// стрелочная функция для обработчиков событий (предотвращение потери контекста)
	handleClick = () => {
		alert(this.state.name);
	}


mounting (метод жизненного цикла)
	componentDidMount() {
        // новое свойство, не видно в render
        this.info = 'Start Variable';
        // вызов метода
        this.changeName('Jack');
	}


unmounting (метод жизненного цикла)
	componentWillUnmount() {
		//
	}


render
	render() {
		return (
			<h2>
				<ComponentA param="data" />
				{this.state.name}
				{this.props.age}
				<button onClick={this.handleClick}>Ok</button>
			</h2>
		);
	}

}

render
ReactDOM.render(
	<App age="25" />,
	document.getElementById('root')
);
