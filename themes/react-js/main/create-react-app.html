Create React App

Getting Started Create React App Available Scripts
  • Create React App - создание одностраничных приложений, предоставляет команды для сборки фронтенда
  • «Под капотом» используются Babel и Webpack, eslint и т.д.

npx create-react-app my-app # пакеты
cd my-app                   # выбор директирии
npm start                   # start
npm run build               # создаст оптимизированную сборку приложения в папке build

Пример

./public/index.html


<div id="root"></div>

./src/index.js


import React from 'react';
import ReactDOM from 'react-dom';
import './index.css';
import App from './components/App/App';

ReactDOM.render(<App />, document.getElementById('root'));

./src/components/App/App.js


import React from 'react';
import logo from '../../assets/logo.svg';

export default class App extends React.Component {
	render() {
		return (
			<div>
				<img src={logo} />
			</div>
		);
	}
}
