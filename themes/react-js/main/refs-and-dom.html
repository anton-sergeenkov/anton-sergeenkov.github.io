Рефы и DOM

  • Рефы дают возможность получить доступ к DOM-узлам или React-элементам, созданным в рендер-методе. Используется когда требуется императивно изменить дочерний элемент, обойдя обычный поток данных
  • Примеры использования
    • Управление фокусом, выделение текста или воспроизведение медиа
    • Императивный вызов анимаций
    • Интеграция со сторонними DOM-библиотеками

import React, { Component } from 'react';

export default class App extends Component {
	constructor(props) {
		super(props);
		this.domElement     = React.createRef(); // ref для dom элемента
		this.classComponent = React.createRef(); // ref для классового компонента
	}
	handleClick = () => {
		const nodeDom       = this.domElement.current;
		const nodeComponent = this.classComponent.current;
		console.log(nodeDom, nodeComponent);
	}
	render() {
		return (
			<div>
				<div ref={this.domElement}>DOM element</div>
				<Child ref={this.classComponent} />
				<button onClick={this.handleClick}>Button</button>
			</div>
		);
	}
}
class Child extends Component {
	render() {
		return (<div>Class Component</div>);
	}
}
