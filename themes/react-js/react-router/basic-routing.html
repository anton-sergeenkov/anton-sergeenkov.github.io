Basic Routing

  • BrowserRouter - router. Использует html5 history api и следит за тем чтобы UI был синхронизирован с тем что написано в адресной строке
  • Route - отображение UI, слелит чтобы совпадали пути между компонентом и адресной строкой
  • Link - видоизмененная ссылка под React для редиректа на нужную страницу. Предотвращает обновление страницы, в отличии от обычной ссылке
  • exact - компонент будет рендериться только по указанному адресу, а не по всем адресам, связанным с ним
  • Router и Route технически они являются компонентами, но сами не создают DOM-элементов

import React, {Component} from 'react';
import { BrowserRouter as Router, Route, Link } from 'react-router-dom';

const Home    = () => ( <h2>Home</h2> )
const About   = () => ( <h2>About</h2> )
const Contact = () => ( <h2>Contact</h2> )

export default class App extends Component {
	render() {
		return (
			<Router>
				<div>
					<Link to="/">Home</Link>
					<Link to="/about/">About</Link>
					<Link to="/contact/">Contact</Link>

					<Route path="/" exact component={Home} />
					<Route path="/about" component={About} />
					<Route path="/contact" component={Contact} />
				</div>
			</Router>
		);
	}
}

Настройка сервера Webpack

  • При переходе по Link изменяется url, но физически этой страницы нет. Поэтому нужно настроить сервер
webpack.config.js
module.exports = {
    output: {
        filename: 'bundle.js',
        path: path.resolve(__dirname, 'build'),
		
		publicPath: '/'
		
	},
	devServer: {
		contentBase: './dist',
		
		historyApiFallback: {
			disableDotRule: true
		},
		
		stats: 'errors-only',
		port: 3000
	}
}
