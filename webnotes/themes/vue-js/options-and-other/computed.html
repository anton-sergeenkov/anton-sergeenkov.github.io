Computed Properties

  • Computed Properties - вычисляемые (обработанные) свойства применяются когда есть зависимость переменных от других переменных

<div id="app">
	<input type="text" @input="increment($event.target.value)">
	<p>{{value}}</p>
	<p>{{doubleValue}}</p>
</div>


var app = new Vue({
	el: '#app', 
	data: {
		value: 1
	},
	methods: {
		increment(value) {
			this.value = value
		}
	},
	computed: {
		doubleValue() {
			return this.value * 2
		}
	}
});

Пример #1. При вводе более 10 символов, скрывается кнопка


<div id="app">
	<form action="">
		<span>{{ count }} / 10</span>
		<textarea v-model="text"></textarea>
		<button v-if="count <= 10">Отправить</button>
	</form>
</div>


var app = new Vue({
	el: '#app',
	data: {
		text: ''
	},
	computed: {
		count() {
			return this.text.length
		}
	}
})
