Фильтры


<div id="app">
	<p>{{ message | lowercase }}</p>
	<p>{{ message | uppercase }}</p>
	<p>{{ message | filter1 | filter2 }}</p>
</div>



// глобальный фильтр
Vue.filter('uppercase', function(value){
	return value.toUpperCase()
});

var app = new Vue({
	el: '#app', 
	data: {
		message: 'Hello'
	},
	// локальный фильтр
	filters: {
		lowercase(value) {
			return value.toLowerCase()
		}
	}
});
