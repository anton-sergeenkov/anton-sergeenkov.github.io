Методы-наблюдатели (watch)

Наблюдение за одним свойством


<input type="text" v-model="messText">


// при изменении свойства messText, вызывается messText()
var app = new Vue({
    el: '#app',
    data: {
        messText: null
    },
    watch: {
        messText() {
            console.log(this.messText);
        }
    }
})
