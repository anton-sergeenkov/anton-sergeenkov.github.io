Передача событий между компонентами

Прослушиване событий


<app-child @close="modal = false"></app-child>   <!-- родительский компонент -->
<button @click="$emit('close')"></button>        <!-- дочерний компонент -->


<app-child @close="modal = $event"></app-child>  <!-- родительский компонент -->
<button @click="$emit('close', true)"></button>  <!-- дочерний компонент -->
