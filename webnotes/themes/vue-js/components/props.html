Передача переменных между компонентами (props)

Родительский компонент


<div id="app">
	<task message="Hello" :name="userName"></task>
</div>


message="Hello"   # передача текста из html
:name="userName"  # передача переменной из app

Дочерний компонент


Vue.component('task', {
	template: '<p>{{ message + name }}</p>',
	props: ['message', 'name']
})
var app = new Vue({
	el: '#app',
    data: {
        userName: 'Tony'
    }
})

Варианты props


props: ['message', 'name']
props: {
	message: String
}
props: {
    count: {
        type: Number,    // задание типа
        default: 77,     // значение по умолчанию
        required: true   // обязательное значение
    }
}

Значением type может быть один из следующих нативных конструкторов: String Number Boolean Array Object Date Function Symbol
