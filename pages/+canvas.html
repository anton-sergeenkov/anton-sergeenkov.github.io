Основы

Выборка и получение контекста


var canvas = document.getElementById('canvas');
var ctx = canvas.getContext('2d');
canvas.width = 200;
canvas.height = 200;
//canvas.width = window.innerWidth;

Общие свойства


ctx.fillStyle = 'yellowgreen';
ctx.strokeStyle = 'black';

Объекты

Rectangle


ctx.fillRect(0,0,100,100);   // fill
ctx.strokeRect(0,0,100,100); // stroke

Arc


ctx.beginPath();
ctx.arc(100, 100, 50, 0, Math.PI*2, false);
ctx.closePath();
ctx.fill();   // fill
ctx.stroke(); // stroke



arc();
1 - координаты центра окружности (x)
2 - координаты центра окружности (y)
3 - радиус
4 - start angle (справа)
5 - end angle (360deg = pi*2)
6 - рисование по часовой (false) / против (true) часовой стрелки

Line, Path


ctx.beginPath();

ctx.moveTo(100, 100);
ctx.lineTo(100, 50);
ctx.lineTo(200, 100);
ctx.closePath();

ctx.moveTo(10, 10);
ctx.lineTo(70, 10);
ctx.lineTo(10, 70);

ctx.fill();
ctx.stroke();

Gradient


var grad = ctx.createLinearGradient(0, 100, 100, 0);
grad.addColorStop(0, 'green');
grad.addColorStop(1, 'blue');
ctx.fillStyle = grad;
ctx.fillRect(50, 50, 100, 100);

Text


ctx.textAlign = 'center';
ctx.font = '22px Arial';
ctx.fillText('Hello World', canvas.width/2, canvas.height/2); // по центру

Text Gradient


var grad = ctx.createLinearGradient(0, 0, 150, 0);
grad.addColorStop(0, 'magenta');
grad.addColorStop(0.5, 'blue');
grad.addColorStop(1, 'red');
ctx.fillStyle = grad;

ctx.font = '22px Arial';
ctx.fillText('Hello World', 20, 35);

Остальное


ctx.clearRect(0, 0, canvas.width, canvas.height); // очистка холста
ctx.scale(2,2);                                   // масштабирование (x,y)
ctx.rotate(10 * Math.PI/180);                     // поворот на 10deg
